/**
        Ejemplo de un procedimiento almacenado que realiza una consulta SELECT
        Comentando un procedimiento almacenado en SQL Server
   */
   CREATE OR ALTER PROCEDURE [dbo].[SP_EJEMPLO_SELECT]
   @DNI VARCHAR(10)
   AS
   BEGIN 
         SELECT NOMBRE, APELLIDO, FECHA_NACIMIENTO
         FROM   USUARIOS
         WHERE  DNI = @DNI
    END

    -- Comentario de una sola lÃ­nea en SQL

    SELECT COUNT(*) AS TOTAL_USUARIOS
    FROM   USUARIOS
    WHERE  FECHA_REGISTRO >= '2023-01-01';
